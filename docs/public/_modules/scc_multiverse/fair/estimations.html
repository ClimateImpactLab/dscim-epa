[docs]def estimate_damages_betas(damage_function,
                           type_estimation,
                           return_preds=True,
                           base_year=2010,
                           interp_year=2085,
                           year_start_pred=2100,
                           year_end_pred=2300):
    """ Estimate global damages betas using global climate anomalies

    This function will calculate a quadratic fit of damages across IAM/GCM/RCP
    using model temperature anomalies from CIMP5. The model to be estimated
    is the following: 

    ..math::
        Damages_{gcm, rcp, iam} = \beta_{0} + \beta_1 * temperature + \beta2 * 
        temperature^{2} + \varepsilon

    This model is estimated for each year in our timeframe (2010 to 2099) and
    using a rolling window across time (5-yr by default). For the years between
    2099 and 2300, we use a linear extrapolation of the quadratic fit of
    damages starting on `interp_year_start` (set as 2085 as the default). For
    this extrapolation we use a linear temporal interaction with the pre-2100
    estimated damages. 

    Regression coefficients for each model are temporally stored and then used
    to estimate predicted damages damages by each quarter-degree
    celsius (C). Hence, we will return a `pd.DataFrame` with predicted damages
    for each year over an array of 40 different temperatures.

    Options are available to change the size of the rolling window, as well as
    the prediction years and base years. 

    Parameters
    ----------
    damage_function : pd.DataFrame 
        Model/Batch damage realizations indexed by temperature anomalies. 
    type_estimation : str
        Type of model use for damage fitting. `ols` and `quantile` are the two
        available options. 
    return_preds : bool
        If `True` return a data frame with predictions. Otherwise, just return
        yearly betas. 
    base_year : int
        Base year to use interpolation. 2010 is the set default. 
    interp_year : int
        Year pre-2100 to fit damages used for linear extrapolation. 
        i.e. If `interp_year=2085`, data from years 2085 to 2099 will be
        used to extrapolate data to years post-2100. 2085 is the default.
    year_start_pred : int
        Start interpolation. Default is 2100
    year_end_pred : int
        End interpolation. Default is 2300

    Returns
    -------

    pd.DataFrame
        A DataFrame containing yearly predictions of damages per each
        quarter-degree starting in 2010 and ending in `year_end_pred`.
        If `return_preds=False`, the the function will return a DataFrame
        with estimation betas per year.
    """

    # Convert to dask arrays to delayed
    damage_function_delayed = dd.from_pandas(damage_function, npartitions=4)

    list_outputs = []
    for year in range(2015, 2100):
        time_window = range(year - 2, year + 3)
        df = damage_function_delayed[damage_function_delayed.year.isin(
            time_window)]
        params = modeler(df=df,
                         formula='damages ~ anomaly + np.power(anomaly, 2)',
                         type_estimation=type_estimation)
        params = params.assign(year=year)
        list_outputs.append(params)

    # Concatenate results
    output_compute = dask.compute(*list_outputs)
    outputs = pd.concat(output_compute)

    # Linear-extrapolation for post-2100 years
    df_interp = damage_function_delayed[
        damage_function_delayed.year >= interp_year]
    df_interp = df_interp.assign(year_rebase=df_interp.year - base_year)

    params = modeler(
        df=df_interp,
        formula='''damages ~ anomaly +
        np.power(anomaly, 2) +
        year_rebase*anomaly +
        year_rebase*np.power(anomaly,2)
        ''',
        type_estimation=type_estimation,
        var_names=['cons', 'anomaly', 'anomaly2', 't', 't*anomaly', 't*anomaly2']).\
        compute()

    extrapolation_year = []
    if type_estimation == 'ols':
        for year in range(year_start_pred, year_end_pred + 1):
            params_df = params.assign(
                year=year,
                cons=params['cons'] + params['t'] * (year - base_year),
                beta_1=params['anomaly'] + params['t*anomaly'] * (year - base_year),
                beta_2=params['anomaly2'] + params['t*anomaly2'] *
                (year - base_year))
            extrapolation_year.append(params_df)
    else:
        years_ranges = range(year_start_pred, year_end_pred)
        quantiles = np.arange(0.05, 1, 0.05)
        for year, quantile in product(years_ranges, quantiles):
            params_ = params[params['q'] == quantile]
            params_df = params_.assign(
                year=year,
                cons=params_['cons'] + params_['t'] * (year - base_year),
                beta_1=params_['anomaly'] + params_['t*anomaly'] *
                (year - base_year),
                beta_2=params_['anomaly2'] + params_['t*anomaly2'] *
                (year - base_year))
            extrapolation_year.append(params_df)

    extrapolation_results = pd.concat(extrapolation_year).\
        filter(['cons', 'beta_1', 'beta_2', 'year', 'q'])

    # Return all results
    res = pd.concat([outputs, extrapolation_results])

    # Calculate prediction for each year
    if return_preds:
        preds = get_the_hats(coefs_df=outputs)
    else:
        preds = res

    return preds




[docs]def compute_damages(fair_temperatures, 
                    betas, 
                    anomalies=False):
    '''Estimate time series mortality damages (copy cat version of Climate Team)

    Calculate damages using FAIR anomalies (either control or pulse). This 
    funtion calculates a base period average and calculate temperature anomalies
    using that climatology. These anomalies are later used to estimate damages
    using the betas from the menu option. 

    Parameters
    ----------
    fair_temperature : xr.DataArray or xr.Dataset
        `xarray` object containing FAIR temperatures. These can be either from 
        control or pulses runs. 
    betas : xr.DataSet
        An `xarray` object containing the fit coefficients from the yearly damage 
        functions
    anomalies : bool
        If `True` do  not calculate temperature anomalies. Then
        fair_temperatures must be anomalies.

    Returns
    -------
    damages : xr.DataArray or xr.Dataset
    '''

    BASEPERIOD = [2001, 2010]
    REFERENCE_YEAR = 1765
    START_YEAR = 2015
    PULSE_YEAR = 2020
    PULSE_AMT = 1.0
    MAGNITUDE_OF_DAMAGES = 1

    # CONVERSION is in units of
    # [pulse/tCO2] = [1 pulse/PULSE_AMT GtC * 1 GtC/1e9 tC * 12tC/44tCO2]
    # Therefore, it should be the inverse of any changes to PULSE_AMT
    CONVERSION = 1.0 / PULSE_AMT / 1e9 * 12.011 / 44.0098

    # Calculate base period mean for each RCP and simulation
    if anomalies:
        temperature_anomalies = fair_temperatures
    else:
        base_period_temp = fair_temperatures.\
            sel(year=slice(BASEPERIOD[0], BASEPERIOD[1])).\
            mean(dim=['year'])

        # Calculate anomaly using base period temperature average
        temperature_anomalies = (fair_temperatures - base_period_temp)

    # Calculate damages using the quadratic relationship of anomalies and
    # damages for each discount rate if available.
    damages_fair = (CONVERSION *
        (betas.cons + betas.beta_1 * temperature_anomalies +
         betas.beta_2 * np.power(temperature_anomalies, 2)))

    LOGGER.warning(
        (
            "Ensure that betas, damages, and SCC are all in the same" 
            " dollar units. There is no unit scaling in this function."
        )
    )

    return damages_fair
