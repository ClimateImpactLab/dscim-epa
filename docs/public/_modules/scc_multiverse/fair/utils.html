@dask.delayed
def modeler(df, formula, type_estimation, var_names=None):
    """Wrapper function for statsmodels functions (OLS and quantiles)

    For the quantiles, the function will fit for all quantiles between 0 and 1
    with a 0.25 step.

    Paramters:
    ---------
    df : pd.DataFrame
        A dataframe of prepared damage function. Since formulas are hard to
        get, the function expects standarized column names: `damages` must
        contain all damages coming from the damage function, and `temp` must
        be the name of temperature anomalies from the models.
    formula : str
        R-like formula to start fitting. (i.e. damages ~ temp +
        np.power(temp,2)) defines a quadratic relationship between temperature
        and damages.
    type_estimation : str
        Model to use . 'ols' and 'quadratic' are the one availables. Any other
        passed option will yield a `NonImplementedError`
    var_names : list(str)
        A list with names for the output columns. If `None` it will return
        constant as `const` and the other estimators indexed as `beta_n`, n
        being the column size of the estimation matrix.

    Returns
    ------
    pd.DataFrame
        A dataframe with coefficients by year.
    """

    if type_estimation == "ols":
        mod = smf.ols(formula=formula, data=df).fit()
        params = pd.DataFrame(mod.params).T
        rhs_size = len(mod.params)

    elif type_estimation == "quantreg":
        with warnings.catch_warnings():
            warnings.filterwarnings("ignore")
            mod = smf.quantreg(formula=formula, data=df)
            quantiles = np.arange(0.05, 1, 0.05)

            quantile_results = []
            for quantile in quantiles:
                fit = mod.fit(q=quantile)
                params_quantile = pd.DataFrame(fit.params).T

                params_quantile = params_quantile.assign(q=quantile)
                quantile_results.append(params_quantile)

        # Take last element columns size and add the quantile column
        rhs_size = len(fit.params)
        params = pd.concat(quantile_results)
    else:
        raise NotImplementedError(f"{type_estimation} is not valid option!")

    # Rename betas on the regression formula
    if var_names is None:
        beta_names = [f"beta_{n}" for n in range(1, rhs_size)]
        var_names = ["cons"] + beta_names

    if "q" in params.columns:
        var_names = var_names + ["q"]

    params.columns = var_names

    return params


def get_weights(quantiles):
    """Compute the weights of each quantile regression.

    Note:
    ----
    `quantiles` must be between 0-1!
    """
    quantiles = np.array(quantiles)
    # find midpoints between quantiles
    bounds = np.array([0] + ((quantiles[:-1] + quantiles[1:]) / 2).tolist() + [1])
    if (bounds > 1).any():
        raise RuntimeError("quantiles must be between 0-1")
    weights = np.diff(bounds)
    return weights


def quantile_weight_quantilereg(ds, quantiles=[0.01, 0.05, 0.25, 0.5, 0.75, 0.95]):
    """Produce quantile weights of the quantile regression damages
    """

    qr_quantiles = ds["q"].values
    ds_stacked = ds.stack(obs=("simulation", "q"))
    weights = xr.DataArray(get_weights(qr_quantiles), dims=["q"], coords=[qr_quantiles])
    weights_by_obs = weights.sel(q=ds_stacked.obs.q)
    # these are quantiles of the full uncertainty, weighted by the quantile regression quantiles
    ds_quantiles = impactlab_tools.utils.weighting.weighted_quantile_xr(
        ds_stacked, quantiles, sample_weight=weights_by_obs.values, dim="obs"
    )
    return ds_quantiles


def get_the_hats(coefs_df, min_anomaly=1, max_anomaly=11, step_anomaly=0.25):
    """Calculate predicted damages for an array of specific GMST values for
    a set of regression outcomes.

    Note
    ----
    This functions is assuming a quadratic relationship

    Parameters
    ----------

    coefs_df: pd.DataFrame
        a data frame with yearly coefficients of the fit results from the
        `modeler`. The expected specific namings for the coefficients: `cons`,
        `beta_1`, `beta_2`, and `year`. Thus rename the `pd.DataFrame`
        accordingly. 
    min_anomaly: int
        Minimum value of GMST anonmaly. Default is 1
    max_anomaly: int
        Maximim value of GMST anomaly. Default is 11
    step_anomaly=0.25
        Step to use between `min_anomaly` and `max_anomaly`

    Returns
    -------
    pd.DataFrame
        A dataframe with predicted damages for each year model and for the
        user-defined array os GMST.
    """

    array_coefs = coefs_df[['cons', 'beta_1', 'beta_2']].values
    array_temp = np.arange(min_anomaly, max_anomaly, step_anomaly)

    temps = np.stack(
        [np.ones(len(array_temp)), array_temp, np.power(array_temp, 2)], axis=1
    ).T

    y_hat = array_coefs.dot(temps)

    # Transform to dataframe format
    if "q" in coefs_df.columns:
        df_preds = pd.DataFrame(
            y_hat, columns=array_temp, index=[coefs_df.year, coefs_df.q]
        ).reset_index()

        df_long = pd.melt(
            df_preds, id_vars=["year", "q"], var_name="anomaly", value_name="y_hat"
        ).sort_values(["year", "anomaly"], ascending=True)

    else:
        df_preds = pd.DataFrame(
            y_hat, columns=array_temp, index=coefs_df.year
        ).reset_index()

        df_long = pd.melt(
            df_preds, id_vars="year", var_name="anomaly", value_name="y_hat"
        ).sort_values(["year", "anomaly"], ascending=True)

    return df_long



[docs]def get_data_home(data_home=None):
    """Return a path to the cache directory for example datasets.
    This directory is then used by :func:`load_parameters`.
    """
    if data_home is None:
        data_home = os.environ.get(
            "FAIR_DATA", os.path.join("~", "fair-parameter-data")
        )
    data_home = os.path.expanduser(data_home)
    if not os.path.exists(data_home):
        os.makedirs(data_home)
    return data_home




[docs]def load_parameters(name, cache=True, token=True, data_home=None, **kwargs):
    """Retrieve parameter files from private data repository

    This function provides a cleaner access to remotely stored data for FAIR
    parameters. Instead of relying on the user file system, this function
    requests parameter NetCDF files from a private Gitlab repo.

    Parameters
    ----------

    name: str
        Name of parameter dataset
    token: str
        Gitlab Private personal token. Default is `None`. Will look for the
        environment variable `GITLAB_PERSONAL_TOKEN`.
    cache: boolean, optional
        If True, try to read from the cached data in the environment variable
        `FAIR_DATA`.
    data_home: str, optional
        Directory to cache data. `fair-parameter-data` by default
    kwargs: dict, optional
        Additional values passed to `xarray.open_dataset`.

    Returns
    -------
    xr_array: :class:`xr.Dataset`
    """

    suffix = Path(name).suffix
    api_endpoint = "https://gitlab.com/api/v4/projects/"
    token = os.environ.get("GITLAB_PERSONAL_TOKEN", token)

    path = quote(f"climate/parameters/{name}", safe="")
    params = f"7426733/repository/files/{path}/raw?ref=master"
    full_url = urljoin(api_endpoint, params)

    headers = {
        "PRIVATE-TOKEN": token,
        "User-Agent": "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.14; rv:66.0) Gecko/20100101 Firefox/66.0",
    }

    if cache:
        cache_path = os.path.join(get_data_home(data_home), name)
        if not os.path.exists(cache_path):
            try:
                response = requests.get(full_url, headers=headers)
                if response.status_code == requests.codes.ok:
                    if suffix == ".csv":
                        string = StringIO(response.text)
                        df = pd.read_csv(string)
                        df = df[[c for c in df.columns if "Unnamed" not in c]]
                        df.to_csv(cache_path, index=False)
                    elif suffix == ".nc":
                        with open(cache_path, "wb") as f:
                            for chunk in response.iter_content(chunk_size=1024):
                                if chunk:
                                    f.write(chunk)
                        df = xr.open_dataset(cache_path)
                else:
                    LOGGER.error(f"Request failed with code: {response.status_code}")
            except Exception as e:
                LOGGER.error(f"Connection failed! {e}")

        else:
            if suffix == ".csv":
                df = pd.read_csv(cache_path)
            elif suffix == ".nc":
                df = xr.load_dataset(cache_path)

    return df
