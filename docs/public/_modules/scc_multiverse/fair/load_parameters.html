[docs]def get_filter_mask(path_to_parameters,
                    CLIMATE_V2 = False, 
                    CLIMATE_V2p1 = False, 
                    CLIMATE_V1 = False):

    if CLIMATE_V2 or CLIMATE_V2p1:
        filters_fp = f'{path_to_parameters}/parameter_filters_rwf_tau4_iptcriteria_v2.1_newiptemissions.nc' 
        print(filters_fp)

        with xr.open_dataset(filters_fp) as filters_ds:
            the_mask = (filters_ds.rwf_mask
                             & filters_ds.tau4_mask
                             & filters_ds.ipt_time_to_dT_lt_0_passing_mask)
            return the_mask

    elif CLIMATE_V1:
        # read in old filters:
        filtered_parameter_indices = pd.read_csv(
            f'{path_to_parameters}/filtered_parameter_indices.csv', 
            index_col=0)

        with xr.Dataset(filtered_parameter_indices) as filters_ds:
            filters_ds.rename({'dim_0':'simulation'})
            the_mask = filters_ds.ipt_dT_lt_0
            return the_mask
    else:
        raise NotImplementedError




[docs]def get_parameters(path_to_parameters,
                   filtered=True, 
                   array=True, 
                   CLIMATE_V2 = False, 
                   CLIMATE_V2p1 = False, 
                   CLIMATE_V1 = False):
    """ if array is False, return the xr.DataArray. Haven't used the DataArray version much. """

    if CLIMATE_V2 or CLIMATE_V2p1: 
        with xr.open_dataset(f'{path_to_parameters}/original_parameter_samples_with_rwf_v2_2019-02-01-22-50-59.nc') as params_ds:

            if filtered:
                climate_params = params_ds.drop('rwf').where(
                    get_filter_mask(
                        path_to_parameters=path_to_parameters,
                        CLIMATE_V2=CLIMATE_V2, 
                        CLIMATE_V2p1=CLIMATE_V2p1, 
                        CLIMATE_V1=CLIMATE_V1),
                    drop=True
                ).to_array(dim='parameter')#.T.values
            else:
                climate_params = params_ds.drop('rwf').to_array(dim='parameter')#.T.values

            if array:
                return climate_params.T.values
            else:
                return climate_params

    elif CLIMATE_V1:
        with xr.open_dataset(f'{path_to_parameters}/climate/parameters/original_parameter_samples.nc') as params_ds:

            if filtered:
                climate_params = params_ds.where(
                    get_filter_mask(
                        path_to_parameters=path_to_parameters,
                        CLIMATE_V2=False, 
                        CLIMATE_V2p1=False, 
                        CLIMATE_V1=True)
                ).to_array(dim='parameter')#.T.values
            else:
                climate_params = params_ds.to_array(dim='parameter')#.T.values

            if array:
                return climate_params.T.values
            else:
                return climate_params
    else:
        raise NotImplementedError




[docs]def get_median_climate_params(path_to_parameters, 
                              CLIMATE_V2 = False, 
                              CLIMATE_V2p1 = False, 
                              CLIMATE_V1 = False):

    cp = get_parameters(path_to_parameters=path_to_parameters, 
                        filtered=True, 
                        array=False, 
                        CLIMATE_V2=CLIMATE_V2, 
                        CLIMATE_V2p1=CLIMATE_V2p1, 
                        CLIMATE_V1=CLIMATE_V1)

    return cp.quantile(0.5, dim='simulation')
