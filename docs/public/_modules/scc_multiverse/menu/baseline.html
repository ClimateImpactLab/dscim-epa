    @property
    def global_damages(self):
        """ Adding-up damages

        Baseline damages are just the aggregation of raw damages to the global
        level. 
        """

        collapsed_damages = (self.damages * self.pop).sum(dim=['region'])
        collapsed_damages_df = (
            collapsed_damages.
            to_array().
            to_dataframe(name=f'global_damages_{self.discounting_type}').
            reset_index()
        )
        collapsed_damages_df['ssp-iam'] = (
            collapsed_damages_df.ssp.str.cat(collapsed_damages_df['model'], sep='-')
        )

        return collapsed_damages_df

    @global_damages.setter
    def global_damages(self, ds):
        """ Adding-up damages

        Baseline damages are just the aggregation of raw damages to the global
        level. This additional method allows to pass a new set of collapsed
        dammages with desired valuation aggregated values. 

        Parameters
        ---------
        ds `xr.DataArray` or `xr.Dataset`
            An `xarray` object with aggregated damages to user-desired levels. 

        Returns
        ------
            pd.DataFrame with damages function with temperature anomalies in
            `Climate.gmst`.
        """

        collapsed_damages_df = (
            ds.
            to_array().
            to_dataframe(name=f'global_damages_{self.discounting_type}').
            reset_index()
        )
        collapsed_damages_df['ssp-iam'] = (
            collapsed_damages_df.ssp.str.cat(collapsed_damages_df['model'], sep='-')
        )

        self.collapsed_damages =  collapsed_damages_df

    @cachedproperty
    def fair_marginal_damages(self):
        """ Calculate marginal damages per puff of CO2 using FAIR for the
        baseline option.

        This function takes the coefficients of the estimated responses of
        damages to temeperature anomalies and calculate the predicted damages
        using the FAIR anomalies.
        """

        damages_pulse = compute_damages(self.climate.fair_pulse,
                                        betas=self.damage_function_coefficients)
        damages_control = compute_damages(self.climate.fair_baselines,
                                          betas=self.damage_function_coefficients)

        marginal_damages = (damages_pulse - damages_control).\
            expand_dims({
                'type_fair': ['full_uncertainty']
            }).chunk({'year': 1})

        if self.climate.fair_median_run is not None:
            damages_pulse_median = compute_damages(
                self.climate.fair_median_run.sel(pulse='pulse'),
                betas=self.damage_function_coefficients)
            damages_control_median = compute_damages(
                self.climate.fair_median_run.sel(pulse='rcp'),
                betas=self.damage_function_coefficients)

            median_marginal_damages = (damages_pulse_median -
                                       damages_control_median).\
                expand_dims({
                    'type_fair': ['parameter_median']
                }).chunk({'year': 1})

            computed_damages = xr.concat(
                [marginal_damages, median_marginal_damages], dim='type_fair')
        else:
            computed_damages = marginal_damages

        return computed_damages
