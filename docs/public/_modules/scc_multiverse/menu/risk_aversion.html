[docs]    def ce_cc_calculation(self) -> xr.DataArray:
        """ Calculate Certainty Equivalence for Risk Aversion. 

        This function is calculating the certainty equivalent damages for the
        risk aversion menu options. Notice that this CE calculations
        are aggregation over several dimensions depending on the
        `discounting_type` selected and that might need more computing power than
        usual. 

        Returns
        -------
             xr.DataArray
        """

        if self.discounting_type in ['ramsey', 'constant']:
            ce_array = self.c_equivalence(array=self.consumptions,
                                          dims=['batch'],
                                          func=None)
        elif self.discounting_type == 'wr':
            ce_array = self.c_equivalence(array=self.consumptions,
                                          dims=['batch', 'ssp', 'model'],
                                          func=None)
        else:
            raise NotImplementedError(
                (f'{self.discounting_type} is not a valid option'
                 f'Try any of these: {self.DISCOUNT_TYPES}'
                 )
            )

        ce_array.name = f'ce_cc_{self.discounting_type}'

        return ce_array



[docs]    def ce_no_cc_calculation(self) -> xr.DataArray:
        """ Calculate certainty equivalent consumption without damages for
        Risk aversion

        This function will calculate the CE for risk-aversion menu options for
        the class specified `discount_type`. Just as the CE damages
        calculatuion, the computing times might vary depending on the
        discounting type. Check the `MainRecipe` class documentation for more
        details. 

        Returns
        -------
            xr.DataArray
        """

        if self.discounting_type in ['ramsey', 'constant']:
            ce_array = self.gdppc
        elif self.discounting_type == 'wr':
            ce_array = self.c_equivalence(array=self.gdppc,
                                          dims=['ssp', 'model'],
                                          func=None)
        else:
            raise NotImplementedError(
                (f'{self.discounting_type} is not a valid option'
                 f'Try any of these: {self.DISCOUNT_TYPES}'
                 )
            )

        ce_array.name = f'ce_no_cc_{self.discounting_type}'

        return ce_array

[docs]    def global_damages_calculation(self) -> pd.DataFrame:
        """ Calculate collapsed damages for a desired discount type

        This function just calculate the damages by taking the difference
        between ce_no_cc and ce_cc. The resulting array operation is
        aggregated and multiplied by population or across SSPs if needed by the
        discount type.

        Returns
        --------
            pd.DataFrame
        """

        if self.discounting_type in ['constant', 'ramsey']:
            dams_collapse = (self.calculated_damages * self.pop).sum(dim='region')
            df = (
                dams_collapse.
                to_dataframe(name=f'global_damages_{self.discounting_type}').
                reset_index()
            )

            df['ssp-iam'] = df.ssp.str.cat(df['model'], sep='-')

        elif self.discounting_type == 'wr':
            dams_collapse = (
                (self.calculated_damages * 
                self.pop.mean(dim=['model', 'ssp'])).
                sum(dim='region')
            )
            df = (
                dams_collapse.
                to_dataframe(name=f'global_damages_{self.discounting_type}').
                reset_index()
            )

            df = df.assign(
                ssp = str(list(self.damages.ssp.values)),
                iam = str(list(self.damages.model.values))
            )

        else:
            raise NotImplementedError(
                (f'{self.discounting_type} is not a valid option \n'
                 f'Try any of these: {self.DISCOUNT_TYPES}'
                 )
            )

        return df



[docs]    def global_consumption_calculation(self):
        """ Calculate Global consumption

        This function, as the others in this class calculate the
        `global_consumption` class property depending on the discount_type
        passed to the class instance.

        Here we calculate global_c (Global Consumption in No Climate Change):
        This object is the baseline that we use to compare damages against. In
        other words, this would be the global consumption where the aditional
        climate change has not happened, and it is usually the baseline we use
        to calculate and collapse damages in `RiskAversion.calculdated_damages`
        but aggregated using the same strategy used on the
        `collapsed_damage_calculation` process.

        Returns
        -------
            xr.DataArray
        """

        if self.discounting_type in ['constant', 'ramsey']:
            global_cons_no_cc = self.gdppc.sum(dim=['region'])

        elif self.discounting_type == 'wr':
            global_cons_no_cc = (
                self.ce_no_cc.sum(dim='region') *
                self.pop.mean(dim=['ssp', 'model']).\
                sum(dim=['region'])
            )
        else:
            raise NotImplementedError(
                (f'{self.discounting_type} is not a valid option \n'
                 f'Try any of these: {self.DISCOUNT_TYPES}'
                 )
            )

        # Convert to array in case xarray becames temperamental. This is a hack
        # that need to be changed
        if isinstance(global_cons_no_cc, xr.Dataset):
            global_cons_no_cc = global_cons_no_cc.to_array()

        global_cons_no_cc.name = f'global_cons_{self.discounting_type}'

        return global_cons_no_cc
