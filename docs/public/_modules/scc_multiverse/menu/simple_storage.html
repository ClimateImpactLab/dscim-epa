    """ Access to all climate variables needed during integration

    This class wraps all climate data used during the integration process. Here
    we can just call files from disparate sources just to have everything in
    one place. 

    Attributes
    ---------
    gmst_path: str
        Path to GMST anomalies data for the CIMP5 models using during
        projection
    fair_path: str
        Path to FAIR baseline and pulse runs. The class will find for pulse and
        baseline file using regex: pulse filename must contain 'baseline',
        while baseline filename should contain 'baseline'. If file is not
        found, `fair_baseline_path` and `fair_pulse_filename` will be used.  
    fair_baseline_path: str
        Path to FAIR baseline. `None` is the default. 
    fair_pulse_path: str
        Path to FAIR pulses file. `None` is the default.
    fair_median_path: str
        Path to FAIR median parameter run to give single estimates of SCC.
        `None` is the default.

    Notes
    ------
    This `Climate` object will contain different climate datasets:
        - gmst: GMST anomalies for CIMP5 models plus surrogates
        - fair: FAIR baseline runs
    """
    def __init__(self,
                 gmst_path,
                 fair_path=None,
                 fair_median_path=None,
                 fair_baseline_path=None,
                 fair_pulse_path=None):
        self.gmst_path = gmst_path
        self.fair_path = fair_path
        self.fair_baseline_path = fair_baseline_path
        self.fair_pulse_path = fair_pulse_path
        self.fair_median_path = fair_median_path

    @cachedproperty
    def gmst(self):
        """ Cached GMST anomalies
        """
        gmst = pd.read_csv(self.gmst_path)

        if 'temp' in gmst.columns:
            gmst = gmst.rename(columns={'temp': 'anomaly'})

        return gmst

    @cachedproperty
    def fair_baselines(self):
        """ Cached FAIR run baseline
        """

        if self.fair_path is None:
            fair_base = xr.open_mfdataset(self.fair_baseline_path,
                                          parallel=True,
                                          chunks={'year': 1})
        else:
            fair_base = xr.open_mfdataset(self.fair_path,
                                          parallel=True,
                                          chunks={
                                              'year': 1
                                          }).sel(runtype='control')

        try:
            fair_base_filter = fair_base.where(fair_base.simulation.isin(
                self.fair_pulse.simulation.values),
                                               drop=True)
        except AttributeError:
            LOGGER.error('simulation dimension not present in FAIR file')

        return fair_base_filter

    @cachedproperty
    def fair_pulse(self):
        """ Cached FAIR run baseline
        """

        if self.fair_path is None:
            fair_pulse = xr.open_mfdataset(self.fair_pulse_path,
                                           parallel=True,
                                           chunks={'year': 1})
        else:
            fair_pulse = xr.open_mfdataset(self.fair_path,
                                           parallel=True,
                                           chunks={
                                               'year': 1
                                           }).sel(runtype='pulse')

        return fair_pulse

    @cachedproperty
    def fair_median_run(self):
        """ Cached FAIR median parameter run including both control and
        baseline
        """

        fair_median = xr.open_mfdataset(self.fair_median_path,
                                        parallel=True,
                                        chunks={'year': 1})
        return fair_median




[docs]class EconVars:
    """ Access to all Economic variables

    This class contains all economic data needed to calculate consumptions and
    do other operations throughout the menu options. `EconVars` will read all
    SSP data and read GDP and population data at the IR/SSP/IAM level. Since
    some menu options rely on number beyond the SSP predictions, we use a
    linear interpolation of population and GDP at the IR level using a
    user-defined period for interpolation. 

    The period of interpolation, defined by the range of `ext_start_year` and
    `ext_end_year`, will select a subset of data to extrapolate values of
    population and income at the IR level. This extrapolation can either be a
    linear 1D extrapolation (just like np.interp) in time, or calculate out of
    sample values assuming a constant growth rate across IR and different
    scenarios (i.e. SSP/IAM) and calculate. The user can change this behavior
    by using the option `ext_method`. 

    Attributes
    ---------
    econ_path str 
        Path to eoconmic data in NetCDF format. 
    ext_subset_start_year int
        Start time for extrapolation start. Default is 2085
    ext_subset_end_year int
        End time for extraplation subset. This is also used as the first year
        for interpolation. Default is 2100
    ext_end_year int
        Time to end the extrapolation period. Default is 2300
    ext_method str
        Extrapolation method: 
            - `'constant'`: constant growth
            - `'linear'`: linear extrapolation
    Notes
    ------
    An `EconVars` object contains different economic variables as class
    properties. You can get:
        - gdp: GDP per capita for each IR/SSP/IAM
        - pop: Population for each IR/SSP/IAM
    """

    def __init__(self, 
                 path_econ,
                 ext_method,
                 eta=2,
                 ext_subset_start_year=2085,
                 ext_subset_end_year=2100,
                 ext_end_year=2300):
        self.path = path_econ
        self.ext_end_year = ext_end_year
        self.ext_subset_start_year = ext_subset_start_year
        self.ext_subset_end_year = ext_subset_end_year
        self.method = ext_method
        self.eta = eta

    @cachedproperty
    def raw_inputs(self):
        """ Load raw values from SSP files
        """

        # load_inpts loads econ data and life expectancies. For a future
        # version we need to clear this out and only recover the economic data
        # not the mortality valuation one. 
        raw = load_inputs(self.path,
                          complete=True,
                          variables=['gdp', 'pop'])
        return raw


[docs]    @staticmethod
    def extrapolate(xr_array, 
                    start_year, 
                    end_year, 
                    interp_year,
                    method,
                    var=None):
        """ Extrapolate values from xarray object indexed in time

        Extrapolate values in xr.DataArray. This function uses the `interp`
        option in xarray objects. The extrapolation can either be linear or
        constant by using the range of years between `start_year` and
        `end_year`. This extrapolation can be linear assuming a linear
        extrapolation rate, or a constant

        Parameters:
        ----------
        xr_array xr.Dataset
            Array with a `'year'` dimension
        start_year int
            Extrapolation subsample start time
        end_year int
            Extrapolation subsample end time
        interp_year int
            End of extrapolation. The function will calculate values from the
            range given by the `end_year` and the `interp_year`.
        var str
            If the `xr_array` is a Dataset, this defined the variable within
            the object. `None` is the default value. 
        method str
            Interpolation method. Valid values are `constant` and `linear`
        """

        if isinstance(xr_array, xr.Dataset) and var is None:
            raise ValueError("Need to pass a var value to subset the xr.Dataset")

        if method == 'linear':
            if var is not None:
                interp_array = (
                    xr_array[var].sel({'year': slice(start_year, end_year)}).
                    interp(year=np.arange(end_year + 1, interp_year + 1 ),
                           method='linear',
                           kwargs={'fill_value': 'extrapolate'}
                           )
                )
            else:
                interp_array = (
                    xr_array.sel(year=slice(start_year, end_year)).
                    interp(year=np.arange(end_year + 1, interp_year + 1 ),
                           method='linear',
                           kwargs={'fill_value': 'extrapolate'}
                           )
                )

        elif method == 'constant':
            growth_rate = xr_array.diff('year') / xr_array.shift(year=1)

            if var is not None:
                subset_data = growth_rate[var].sel(year=slice(start_year, end_year))
            else:
                subset_data = growth_rate.sel(year=slice(start_year, end_year))

            # Calculate mean rate over the interpolation period and return a
            # np.array
            mean_growth_rate = subset_data.mean(dim='year').values

            # Repeat mean growth rate across time. Here we create a new index
            # and fill it with the repeat function. The indexing is tricky. 
            repeated_growth = (
                mean_growth_rate.reshape(mean_growth_rate.shape + (1,)).
                repeat(len(range(end_year + 1, interp_year + 1 )),
                       len(mean_growth_rate.shape))
            )

            # For each year we calculate the cummulative growth rate (1+r)
            cumulative_growth = np.multiply.accumulate((1 + repeated_growth),
                                                       len(mean_growth_rate.shape)) 
            last_value = (
                xr_array.sel(year=slice(start_year, end_year)).
                isel(year=-1)
            )

            # Set dict of coords to generalize the xarray transformation using
            # growth as a template

            dict_coords = dict(growth_rate.coords)
            dict_coords.pop('year', None)
            coords_tuples = [(k, v.values) for k, v in dict_coords.items()]
            coords_tuples.append(('year', np.arange(end_year + 1, interp_year + 1)))


            # Set shape
            shapes_base = [v.values.shape[0] for k, v in dict_coords.items()]
            shapes_base.append(len(np.arange(end_year + 1, interp_year + 1)))

            growth_array = xr.DataArray(cumulative_growth.reshape(shapes_base),
                                        coords=coords_tuples)

            # Dot product of last year value and the array of growth rates
            interp_array =  growth_array * last_value

        else:
            ValueError(f"{method} is not a valid interpolation method")

        return interp_array

    @cachedproperty
    def stream_discount_factors(self):
        """ Stream of discount rates

        To calculate discountted damages we use the growth rates to calculate a
        yearly stream of discount factors. In the case of risk aversion, these rates
        (delta_{jt}, as in the implementation memo), are indexed by default on
        ssp and time. In the case of Weitzman-Ramsey discounting, the user can
        use `xarray` to calculate the average across SSPs.

        Since we only have growth for the periood covered by the GCMs (2015,
        2099), we assign a constant growth using the same years used for damage
        extrapolationo during the FAIR step. This is needed to capture the
        complete stream of damages.

        Before any calculation we need to fix the 5-year period interpolation
        process within the data. For this process we will follow this steps:
        1. Select all data in 5-year steps
        2. Calculate the log of the values in levels
        3. Linearly interpolate to fill data between the 5-year steps. Hence
        have a complete dataset between 2015 and 2099
        4. Calculate growth rates after transforming the data back to levels
        5. Calculate growth in the last year and keep it constant until 2300.

        Returns
        -------
        `xarray.DataArray`
            Discount rates indexed by year and SSP
        """

        # Calculate global GDPpc
        global_gdppc = self.gdp.sum(dim='region')/self.pop.sum(dim='region')

        # Subset data in 5 year steps and calculate the log of data in levels 
        multiple_year = [year for year in global_gdppc.year.values if year % 5 == 0]
        subset_gdppc = np.log(global_gdppc.sel(year=multiple_year))

        # Linear extrapolation to fill data between 00s and 05s
        interpolated_gdppc = subset_gdppc.interp(
            year=[year for year in global_gdppc.year.values if year not in multiple_year],
            method='linear'
        )
        complete_gdppc = (
            xr.concat([subset_gdppc, interpolated_gdppc], dim='year').
            sortby('year')
        )
        # Turn back to levels and calculate growth rates
        global_gdppc_int = np.exp(complete_gdppc)
        growth_rates = global_gdppc_int.diff('year') / global_gdppc_int

        # Take growth in last year and extrapolate
        growth_rates_base = growth_rates.isel(year=-1).values
        growth_rates_ext = growth_rates.interp(
            year=np.arange(self.ext_subset_end_year + 1, self.ext_end_year + 1),
            method='linear',
            kwargs={'fill_value': growth_rates_base}
        )

        growth_rates_complete = xr.concat([growth_rates, growth_rates_ext],
                                          dim='year')

        # Calculate stream of discount factors
        stream_rates = (
            np.divide(1, np.multiply.accumulate((growth_rates_complete.values + 1), 2))
        )

        stream_rates_array = xr.DataArray(
            stream_rates,
            coords=[
                ('ssp', growth_rates_complete.ssp.values),
                ('model', growth_rates_complete.model.values),
                ('year', growth_rates_complete.year.values),
            ])

        return stream_rates_array




[docs]class StackedDamages:
    """ Wrapper to sector stacked damages

    This class will take wrap the stacking of one (or several) sector damages
    coming from the climate-projection system. This function will read all
    monetized damages in NetCDF format and combine them by chunks to allow
    operation in larger-than-memory databases. 

    It is esential that the user start a `dask.distributed.Client` before
    calling this class. The `Client` object will read data by chunk and run the
    operations across the class methods. If no `Client` is present, the
    operation will be halted. This is assuming the damage data can't be loaded
    into memory (which is actually a good practice). 

    This class is not supposed to be instantiated directly by the user, but
    rather by accessed using the `Menu` classes instead.

    Attributes
    ---------
    sector_path : str
        Path to project data
    subset_damages : dict
        Dictionary with optional keys to subset damages
    econ_vars : scc_multiverse.simple_storage.EconVars cls
        An `EconVars` object with the correct paths to data sources
    climate_vars : scc_multiverse.simple_storage.Climate cls
        An `Climate` object with the correct paths to data sources
    cons_clip : int
        A integer value to bottom clip negative consumptions
    subset_dict dict
        A dictorionary with coordinate values to filter data. The keys of the
        dictorinary take the values of the coordinate and the values take the
        filtering values:
        ```
        d = {'year': range(2010, 2015),
            'iam': 'OECD GDP'
            }
        ```

    Methods
    -------
    cut
        Cut an xarray object using a dict object
    bottom_coding
        Asssign a user-defined value to negative consumptions
    """

    chunks = {'batch': 1, 'model': 1, 'rcp': 1, 'ssp': 1, 'gcm': 1}

    def __init__(self,
                 sector_path,
                 econ_vars,
                 climate_vars,
                 cons_clip=None,
                 subset_dict=None):

        self.path = sector_path
        self.cons_clip = cons_clip
        self.subset_dict = subset_dict
        self.econ_vars = econ_vars
        self.climate = climate_vars

    @property
    def files(self):
        """ List damage files in `StackedDamages.path`. 

        The function will find first NetCDF (.nc4) files, and Zarr (.zarr)
        files.
        """

        _files = list(pathlib.Path(self.path).rglob('*.nc4'))

        if not _files:
            _files = list(pathlib.Path(self.path).rglob('*.zarr'))

        return _files

    @property
    def damages(self):
        """ Lazy read files and persist data to memory
        """

        xr_data = xr.open_mfdataset(self.files,
                                    chunks=self.chunks,
                                    parallel=True)

        if self.subset_dict is not None:
            xr_data = self.cut(xr_data)

        return xr_data


[docs]    def cut(self, xr_array):
        """ Slice array to desired subset

        Parameters
        ----------
        xr_array :  xr.Dataset or xr.Dataarray
            An xarray object

        Returns
        -------
        xr.Dataset or xr.Dataarray
            `xarray` object filtered using the dict defined in the class:
            `self.subset_dict`
        """

        valid_keys = {
            key: self.subset_dict[key]
            for key in self.subset_dict if key in xr_array.coords
        }

        xr_data = xr_array.sel(valid_keys)

        return xr_data


    @cachedproperty
    def gdp(self):
        if self.subset_dict is not None:
            gdp_cut = self.cut(self.econ_vars.gdp)

            if 'ssp' not in gdp_cut.dims:
                gdp_cut = gdp_cut.expand_dims(ssp=self.subset_dict['ssp'])

        else:
            gdp_cut = self.econ_vars.gdp

        return gdp_cut

    @cachedproperty
    def pop(self):
        if self.subset_dict is not None:
            pop_cut = self.cut(self.econ_vars.pop)
        else:
            pop_cut = self.econ_vars.pop

        return pop_cut 

    @cachedproperty
    def gdppc(self):
        if self.subset_dict is not None:
            pop_cut = self.cut(self.econ_vars.gdppc)
        else:
            pop_cut = self.econ_vars.gdppc

        return pop_cut 

    @cachedproperty
    def stream_discount_factors(self):
        if self.subset_dict is not None:
            stream_cut = self.cut(self.econ_vars.stream_discount_factors)
        else:
            stream_cut = self.econ_vars.stream_discount_rates

        return stream_cut

    @property
    def consumptions(self):
        """ Calculate consumption using damages

        This function just takes the damages stored in the class and substract
        them from the gdp for each combination of coordinates. This operation
        is broadcasted using xarray, so we can be certain that IRs and other
        coordinates are matching during the operation

        Returns
        ------
        xr.Dataarray
            Cached `xarray` object with consuption 
        """

        client = _get_global_client()
        if client is None:
            LOGGER.info(f'WARNING! No Dask client available!')

        cons = (self.gdppc - self.damages).to_array()

        if self.cons_clip is None:
            # Burundi's 2010 GDP per capita. This is the lowest GDP per capita in
            # capita in 2010 for the wholew world using the WorldBank dataset
            bottom_coding_gdppc = 234.235646874999
            cons_clip = cons.where((np.isnan(cons) | (cons > 0)), bottom_coding_gdppc)
        else:
            cons_clip = cons.where((np.isnan(cons) | (cons > 0)), self.cons_clip)

        return cons_clip
